
services:
  apiescuela-backend: 
    container_name: apiescuela-backend
    build: 
      context: .
      dockerfile: Dockerfile
      target: backend
    image: golang:1.24-alpine
    command: go run main.go
    restart: always
    volumes:
      - ./ApiEscuela:/home/app
      - ./ApiEscuela/assets/images:/home/app/assets/images
    ports:
      - '3000:3000'
    env_file:
      - .env
    environment:
      - WHATSAPP_SERVICE_URL=http://wa-node-service:3001
    depends_on:
      - wa-node-service
    networks:
      - escuela_network

  wa-node-service:
    container_name: wa-node-service
    image: node:22-alpine
    working_dir: /app
    command: sh -c "apk add --no-cache chromium && npm install && npm start"
    restart: always
    volumes:
      - ./wa-node-service:/app
      - ./wa-node-service/.wwebjs_auth:/app/.wwebjs_auth
      - /app/node_modules
    environment:
      - PORT=3001
      - PUPPETEER_SKIP_CHROMIUM_DOWNLOAD=true
      - PUPPETEER_EXECUTABLE_PATH=/usr/bin/chromium-browser
    networks:
      - escuela_network

  frontedescuela-frontend:
    container_name: frontedescuela-frontend
    image: node:22-alpine
    working_dir: /app
    command: sh -c "npm install && npm run dev -- --host 0.0.0.0"
    restart: always
    volumes:
      - ./FrontedEscuela:/app
      - /app/node_modules
    ports:
      - "5173:5173"
    environment:
      - VITE_API_URL=http://localhost:3000
      - CHOKIDAR_USEPOLLING=true
    depends_on:
      - apiescuela-backend
    networks:
      - escuela_network

networks:
  escuela_network:
    driver: bridge
